char	*remove_qoutes(char *str)
{
	char	*value;
	char	*s;
	int		c[2];
	t_lexer	*lexer;
	char	*tmp;

	lexer = init_lexer(str);
	value = ft_calloc(1, sizeof(char));
	c[0] = '"';
	c[1] =  '\'';
	int	i = 0;
	while (lexer->c != '\0')
	{
		if(lexer->c == c[0] || lexer->c == c[1])
		{
			if(i == 0)
			{
				i++;
				c[0] = lexer->c;
				c[1] =  lexer->c;
			}
			else
			{
				c[0] = '"';
				c[1] =  '\'';
			}
			lexer_advance(lexer);
		}
		s = lexer_get_current_char_as_string(lexer);
		tmp = value;
		value = ft_realloc(value, (ft_strlen(value)
					+ ft_strlen(s) + 1) * sizeof(char));
		free(tmp);
		ft_strcat(value, s);
		free(s);
		lexer_advance(lexer);
	}
	free(lexer);
	return (value);
}
